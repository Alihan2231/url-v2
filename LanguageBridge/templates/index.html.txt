                    <script>
                    // Form gönderilmeden önce yükleme ekranını göster ve giriş kontrolü yap
                    document.getElementById('url-scan-form').addEventListener('submit', function(e) {
                        e.preventDefault(); // Form gönderimini durdur
                        
                        const urlInput = document.getElementById('url-input');
                        let url = urlInput.value.trim();
                        
                        if (url === '') {
                            alert('Lütfen bir URL giriniz.');
                            return false;
                        }
                        
                        // URL protokolü ekle eğer yoksa
                        if (!url.startsWith('http://') && !url.startsWith('https://')) {
                            url = 'http://' + url;
                            urlInput.value = url; // Protokolü eklenmiş URL'yi forma yaz
                        }
                        
                        // Spinner göster
                        document.getElementById('spinner').style.display = 'block';
                        document.getElementById('scan-button').disabled = true;
                        
                        // Formu gönder
                        this.submit();
                    });
                    </script>
